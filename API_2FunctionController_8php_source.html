97 private function getLastMovieIndex(Room $room, string $date, array $movieIds): int
98 {
99 $previousBusinessDay = $this->getPreviousBusinessDay(Carbon::parse($date));
100
101 $lastFunction = Functions::where('room_id', $room->id)
102 ->whereDate('date', $previousBusinessDay->format('Y-m-d'))
103 ->orderBy('time', 'desc')
104 ->first();
105
106 if (!$lastFunction) {
107 return 0;
108 }
109
110 // Encontrar el índice de la última película en el array de movie_ids
111 $index = array_search($lastFunction->movie_id, $movieIds);
112
113 // Si encontramos el índice, devolver el siguiente. Si no, empezar desde 0
114 return $index !== false ? ($index + 1) % count($movieIds) : 0;

121 {
122 try {
123 $request->validate([

125 'date' => 'required|date',
126 'movie_ids' => 'required|array',
127 'movie_ids.*' => 'exists:movies,id'
128 ]);
129
130 $room = Room::findOrFail($request->room_id);
131 $date = Carbon::parse($request->date);
132
133 // Verificar si es día laborable
134 if (!$this->isBusinessDay($date)) {
135 return response()->json([
136 'success' => false,
137 'message' => 'No se pueden generar funciones para fines de semana',
138 'next_business_day' => $this->getNextBusinessDay($date)->format('Y-m-d')
139 ], 400);
140 }
141
142 $movies = Movie::whereIn('id', $request->movie_ids)->get();
143 $movieIds = $request->movie_ids;
144
145 // Comenzar transacción
146 DB::beginTransaction();
147
148 // Eliminar funciones existentes para esta sala y fecha
149 Functions::where('room_id', $room->id)
150 ->whereDate('date', $date->format('Y-m-d'))
151 ->delete();
152
153 $currentTime = Carbon::parse($date->format('Y-m-d') . ' ' . self::OPENING_TIME);
154 $closingTime = Carbon::parse($date->format('Y-m-d') . ' ' . self::CLOSING_TIME);
155
156 // Obtener el índice de la última película del día laborable anterior
157 $movieIndex = $this->getLastMovieIndex($room, $date->format('Y-m-d'), $movieIds);
158 $scheduledFunctions = [];
159
160 while ($currentTime < $closingTime) {
161 if ($movieIndex >= $movies->count()) {
162 $movieIndex = 0; // Volver al inicio de la lista de películas
163 }
164
165 $movie = $movies[$movieIndex];
166
167 // Calcular fin de la película
168 $endTime = (clone $currentTime)->addMinutes($movie->duration);
169
170 // Si la película termina después de la hora de cierre, verificar si es razonable
171 if ($endTime->format('H:i') > self::CLOSING_TIME) {
172 // Si la película comienza muy tarde, terminar aquí
173 if ($currentTime->format('H:i') > '22:00') {
174 break;
175 }
176 }
177
178 // Crear la función
179 $function = Functions::create([
180 'movie_id' => $movie->id,
181 'room_id' => $room->id,
182 'date' => $date->format('Y-m-d'),
183 'time' => $currentTime->format('H:i:s'),
184 'is_3d' => $room->cinema->has_3d && $movieIndex % 2 == 0, // Alternar 3D si está disponible
185 'price' => $room->cinema->has_3d && $movieIndex % 2 == 0 ? 12.00 : 8.00 // Precio base + extra por 3D
186 ]);
187
188 // Crear asientos para esta función
189 $seatNumber = 1;
190 for ($row = 0; $row < $room->rows; $row++) {
191 for ($col = 0; $col < $room->seats_per_row; $col++) {
192 $function->seats()->create([
193 'number' => $seatNumber,
194 'row' => chr(65 + $row),
195 'status' => Seat::STATUS_AVAILABLE,
196 'price' => $room->price + ($function->is_3d ? 2 : 0)
197 ]);
198 $seatNumber++;
199 }
200 }
201
202 $scheduledFunctions[] = [
203 'id' => $function->id,
204 'movie' => $movie->title,
205 'start_time' => $currentTime->format('H:i'),
206 'end_time' => $endTime->format('H:i'),
207 'is_3d' => $function->is_3d,
208 'sequence_number' => $movieIndex + 1 // Agregar número de secuencia
209 ];
210
211 // Avanzar al siguiente horario (duración + tiempo de limpieza)
212 $currentTime = $endTime->addMinutes(self::CLEANING_TIME);
213 $movieIndex++;
214 }
215
216 DB::commit();
217
218 return response()->json([
219 'success' => true,
220 'data' => [
221 'room' => $room->name,
222 'date' => $date->format('Y-m-d'),
223 'is_business_day' => true,
224 'next_business_day' => $this->getNextBusinessDay($date)->format('Y-m-d'),
225 'next_movie_index' => $movieIndex % $movies->count(),
226 'functions' => $scheduledFunctions
227 ],
228 'message' => 'Horarios generados correctamente'
229 ]);
230
231 } catch (\Exception $e) {
232 DB::rollBack();
233 Log::error('Error en FunctionController@generateSchedule: ' . $e->getMessage());
234 return response()->json([
235 'success' => false,
236 'message' => 'Error al generar horarios: ' . $e->getMessage()
237 ], 500);
