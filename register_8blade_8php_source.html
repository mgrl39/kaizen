148 email: document.getElementById('email').value,
149 password: document.getElementById('password').value,

152 // Iniciar estado de carga

171 if (!response.ok) {

176 errorElement.textContent = data.errors[key][0];

178 }

180 showMessage(false, 'Por favor corrige los errores en el formulario');

183 }

187 if (data.token) {

189 }
190

192 localStorage.setItem('auth_user', JSON.stringify(data.user));
193 }
194
195 setTimeout(() => {

197 }, 2000);
198 }
199 } catch (error) {

202 submitBtn.disabled = false;
