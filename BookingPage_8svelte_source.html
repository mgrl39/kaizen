174 async function handleBooking() {
175 if (isSubmitting) return;
176
177 // Validar que estemos en el paso correcto
178 if (currentStep !== STEPS.SUMMARY) {
179 error = 'Por favor, complete todos los pasos antes de confirmar la reserva';
180 return;
181 }
182
183 isSubmitting = true;
184 error = '';
185
186 try {
187 // Validación de asientos
188 if (!Array.isArray(selectedSeats) || selectedSeats.length === 0) {
189 throw new Error('Debes seleccionar al menos un asiento');
190 }
191
192 // Validación de datos del comprador
193 if (!buyer.name.trim()) {
194 throw new Error('El nombre es obligatorio');
195 }
196 if (!buyer.email.trim() || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(buyer.email)) {
197 throw new Error('El email no es válido');
198 }
199 if (!buyer.phone.trim()) {
200 throw new Error('El teléfono es obligatorio');
201 }
202
203 const headers = {
204 'Content-Type': 'application/json',
205 'Accept': 'application/json'
206 };
207
208 // Solo añadir el token si existe (opcional)
209 const token = localStorage.getItem('token');
210 if (token) {
211 headers['Authorization'] = `Bearer ${token}`;
212 }
213
214 const response = await fetch(`${API_URL}/bookings`, {
215 method: 'POST',
216 headers,
217 body: JSON.stringify({
218 function_id: functionData.id,
219 seats: selectedSeats,
220 buyer: {
221 name: buyer.name.trim(),
222 email: buyer.email.trim(),
223 phone: buyer.phone.trim()
224 }
225 })
226 });
227
228 const result = await response.json();
229
230 if (!response.ok) {
231 if (result.errors) {
232 // Manejar errores de validación específicos
233 const errorMessages = [];
234 if (result.errors['buyer.name']) {
235 errorMessages.push('El nombre es obligatorio');
236 }
237 if (result.errors['buyer.email']) {
238 errorMessages.push('El email es obligatorio y debe ser válido');
239 }
240 if (result.errors['buyer.phone']) {
241 errorMessages.push('El teléfono es obligatorio');
242 }
243 if (result.errors['seats.0']) {
244 errorMessages.push('El asiento seleccionado no es válido');
245 }
246 throw new Error(errorMessages.join('. '));
247 }
248 throw new Error(result.message || 'Error al realizar la reserva');
249 }
250
251 if (!result.success) {
252 throw new Error(result.message || 'Error al realizar la reserva');
253 }
254
255 if (result.success && result.data?.booking?.uuid) {
256 success = true;
257 ticketUrl = result.data.ticket_url || result.data.qr_url;
258 currentStep = STEPS.CONFIRMATION;
259 }
260 } catch (e: any) {
261 console.error('Error en la reserva:', e);
262 error = e.message;
263 success = false;
264 // Si hay un error de validación, volver al paso de contacto
265 if (e.message.includes('obligatorio') || e.message.includes('válido')) {
266 currentStep = STEPS.CONTACT;
267 }
268 } finally {
269 isSubmitting = false;
270 }
