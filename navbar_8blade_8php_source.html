171document.addEventListener('DOMContentLoaded', function() {
172 // --- Gestión de Autenticación en la interfaz (solo para JWT) ---
173 // Solo ejecutamos esta lógica si no hay sesión de Laravel activa
174 if (!document.querySelector('.auth-user.d-none')) {
175 // Función para comprobar si el usuario está autenticado por JWT
176 function checkAuthentication() {
177 const token = localStorage.getItem('auth_token');
178 const authUser = localStorage.getItem('auth_user');
179
180 if (token && authUser) {
181 try {
182 // Intentar parsear la información del usuario
183 const userData = JSON.parse(authUser);
184
185 // Actualizar la interfaz para usuario autenticado
186 document.querySelectorAll('.auth-guest').forEach(el => el.classList.add('d-none'));
187 document.querySelectorAll('.auth-user').forEach(el => el.classList.remove('d-none'));
188
189 // Mostrar el nombre de usuario
190 const usernameDisplay = document.getElementById('username-display');
191 if (usernameDisplay) {
192 usernameDisplay.textContent = userData.username || userData.name || 'Usuario';
193 }
194
195 // Actualizar avatar si existe
196 const userAvatar = document.getElementById('user-avatar');
197 const avatarPlaceholder = document.querySelector('.avatar-placeholder');
198 if (userAvatar && userData.avatar) {
199 userAvatar.src = userData.avatar;
200 userAvatar.classList.remove('d-none');
201 avatarPlaceholder.classList.add('d-none');
202 }
203
204 return true;
205 } catch (e) {
206 console.error('Error parsing auth user data', e);
207 return false;
208 }
209 } else {
210 // Actualizar la interfaz para invitado
211 document.querySelectorAll('.auth-guest').forEach(el => el.classList.remove('d-none'));
212 document.querySelectorAll('.auth-user').forEach(el => el.classList.add('d-none'));
213
214 return false;
215 }
216 }
217
218 // Función para cerrar sesión (JWT)
219 function logout() {
220 // Eliminar el token y datos de usuario
221 localStorage.removeItem('auth_token');
222 localStorage.removeItem('auth_user');
223
224 // Actualizar la interfaz
225 checkAuthentication();
226
227 // Redireccionar al inicio
228 window.location.href = '/';
229 }
230
231 // Comprobar autenticación al cargar la página
232 checkAuthentication();
233
234 // Evento para el botón de cerrar sesión
235 const logoutBtn = document.getElementById('logout-btn');
236 if (logoutBtn) {
237 logoutBtn.addEventListener('click', function(e) {
238 e.preventDefault();
239 logout();
240 });
241 }
242
243 // Escuchar cambios en el almacenamiento (por si se inicia sesión en otra pestaña)
244 window.addEventListener('storage', function(e) {
245 if (e.key === 'auth_token' || e.key === 'auth_user') {
246 checkAuthentication();
247 }
248 });
249 }
250});
251</script>
