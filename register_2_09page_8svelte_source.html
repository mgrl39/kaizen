51 errors.email = $t('emailInvalid');

53
54 // Password validation

58 errors.password = $t('passwordMinLength');

60
61 // Password confirmation

63 errors.passwordConfirmation = $t('passwordsDontMatch');
64 }
65
66 // Birthdate validation is optional now
67 if (birthdate) {
68 const today = new Date();
69 const selectedDate = new Date(birthdate);
70 if (selectedDate >= today) {
71 errors.birthdate = $t('birthdateInvalid');
72 }
73 }

78 async function registerUser(userData: any) {

91 if (!response.ok) {
92 if (data.errors) {

94 const formattedErrors: ErrorsType = {};
95 for (const key in data.errors) {
96 formattedErrors[key] = data.errors[key][0]; // Tomar el primer mensaje de error
97 }
98 return {
99 success: false,
100 message: $t('formErrors'),
101 errors: formattedErrors

105 success: false,
106 message: data.message || $t('registerError')
107 };
108 }
109
110 // Guardar solo el token, no información del usuario
111 if (data.token) {
112 localStorage.setItem('token', data.token);
113 }
114
115 return {
116 success: true,
117 data
118 };
119 } catch (error) {
120 return {
121 success: false,
122 message: $t('connectionError')
