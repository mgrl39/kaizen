43 if (!name) errors.name = $t('nameRequired');

51 if (password !== passwordConfirmation) errors.passwordConfirmation = $t('passwordsDontMatch');
52 // Birthdate validation is optional now

56 if (selectedDate >= today) errors.birthdate = $t('birthdateInvalid');
57 }
58 // Terms validation
